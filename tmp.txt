import { Navigate, useLocation } from "react-router-dom";
import { useAuth } from "./AuthContext";
import useConsent from "../consent/useConsent";
import ConsentModal from "../consent/ConsentModal";

export default function ProtectedRoute({ children, need = [] }) {
  const { initializedSession, loading, user, permissions } = useAuth();
  const { needsConsent, loading: consentLoading, refetchConsent } = useConsent();
  const location = useLocation();

  console.groupCollapsed(
    `%c[ROUTE] ProtectedRoute â†’ ${location.pathname}`,
    "color:#03a9f4;font-weight:bold"
  );
  console.log("initializedSession:", initializedSession);
  console.log("loading:", loading);
  console.log("user:", user);
  console.log("required perms:", need);
  console.log("user perms:", permissions);
  console.groupEnd();

  // 1) Esperando carga inicial de sesiÃ³n
  if (!initializedSession) {
    return (
      <div style={{ padding: 40, fontSize: 18 }}>Cargandoâ€¦ (session)</div>
    );
  }

  // 2) Roles/permisos cargando
  if (loading) {
    return (
      <div style={{ padding: 40, fontSize: 18 }}>Cargandoâ€¦ (roles)</div>
    );
  }

  // 3) No hay usuario â†’ login
  if (!user) {
    console.warn("[ROUTE] âŒ Sin usuario â†’ /login");
    return <Navigate to="/login" replace />;
  }

  // 4) Validar permisos
  if (need.length > 0) {
    const ok = need.every((p) => permissions.includes(p));
    if (!ok) {
      console.warn("[ROUTE] âŒ Permisos insuficientes â†’ /403");
      return <Navigate to="/403" replace />;
    }
  }

  console.log("[ROUTE] âœ” Acceso permitido");
  return children;
}
